resource "kubernetes_persistent_volume_claim" "mobius12-efs" {
  metadata {
    name = var.mobius_persistentVolume_claimName
    namespace  = var.NAMESPACE
  }
  spec {
    access_modes = ["ReadWriteOnce"]
    resources {
      requests = {
        storage = "1Gi"
      }
    }
  }

}
